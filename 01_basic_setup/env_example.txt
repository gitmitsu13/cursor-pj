# 市場情報自動収集・分析システム 環境変数設定例
# このファイルを .env にコピーして、実際の値を設定してください

# NewsAPI設定
# https://newsapi.org/ で無料アカウントを作成してAPIキーを取得
NEWS_API_KEY=your_news_api_key_here

# OpenAI API設定
# https://platform.openai.com/ でアカウントを作成してAPIキーを取得
OPENAI_API_KEY=your_openai_api_key_here

# Gmail設定
# Gmail APIを使用する場合のOAuth2.0認証情報
GMAIL_CLIENT_ID=your_gmail_client_id_here
GMAIL_CLIENT_SECRET=your_gmail_client_secret_here
GMAIL_REFRESH_TOKEN=your_gmail_refresh_token_here

# または、smtplibを使用する場合（Googleアカウントのアプリパスワードが必要）
GMAIL_USERNAME=your_email@gmail.com
GMAIL_APP_PASSWORD=your_app_password_here

# メール送信設定
RECIPIENT_EMAIL=recipient@example.com
SENDER_EMAIL=your_email@gmail.com

# システム設定
LOG_LEVEL=INFO
MAX_NEWS_ARTICLES=10
